// ── Ajouter dans ton Worker mijopayv2 ou créer investor.mijocomplexe.workers.dev ──
 
// Ajouter ce binding dans Cloudflare → Worker → Liaisons :
// CENTRAL_WORKER → Service binding → "central"
 
if (action === "get_investor_data") {
  // Récupérer les données du compte entreprise
  const [userRes, txnRes] = await Promise.all([
 
    env.CENTRAL_WORKER.fetch(new Request(
      `https://central/Utilisateurs?userId=eq.rec6v8Z0K0G4bwbdE&select=Solde,Solde_USD,NomUtilisateur&limit=1`,
      { method: "GET", headers: { "X-MIJO-Bridge-Token": env.MIJO_INTERNAL_SECRET_TOKEN }}
    )),
 
    env.CENTRAL_WORKER.fetch(new Request(
      `https://central/Transactions?select=TypeTransaction,Montant,Date,Statut&order=Date.desc&limit=50`,
      { method: "GET", headers: { "X-MIJO-Bridge-Token": env.MIJO_INTERNAL_SECRET_TOKEN }}
    ))
  ]);
 
  const [userData, txnData] = await Promise.all([userRes.json(), txnRes.json()]);
  const user = userData[0];
 
  // Agréger volume par mois
  const byMonth = {};
  (txnData || []).forEach(t => {
    const m = (t.Date || "").substring(0, 7);
    if (m) byMonth[m] = (byMonth[m] || 0) + parseFloat(t.Montant || 0);
  });
 
  return new Response(JSON.stringify({
    success: true,
    soldeHTG:  user?.Solde     || 0,
    soldeUSD:  user?.Solde_USD || 0,
    nom:       user?.NomUtilisateur,
    updatedAt: new Date().toISOString(),
    chartData: {
      labels: Object.keys(byMonth).sort(),
      volumes: Object.keys(byMonth).sort().map(k => byMonth[k])
    },
    recentTxns: (txnData || []).slice(0, 10)
  }), { headers: { ...corsHeaders, "Content-Type": "application/json" }});
}
